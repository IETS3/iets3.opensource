meta structure machine {
  query calculate(a: number, b: number) => number = a + b + runtime.x
} runtime x: number

machine Computer {
}

assert new Computer(3).calculate(1, 2) equals 6
assert new Computer(3).calculate(1) yields NoSuchMethodError

machine Abakus {
  calculate calcWithZero 0
}

assert new Abakus(3).calcWithZero(2) equals 5
assert new Abakus(3).calculate(1,2) equals 6

machine Spickzettel {
  calculate calcWithOneAndTwo 1, 2
}

assert new Spickzettel(3).calculate(2)  yields NoSuchMethodError
assert new Spickzettel(3).calculate(1,2) equals 6
assert new Spickzettel(3).calcWithOneAndTwo() equals 6
