##  When timer expires, the automatic door controller shall close door until door closing limit sensor is activated

##  When door closing limit sensor is activated, the automatic door controller shall stop door
[] (doorclosinglimitsensorisactivated -> doorstop)

[] (doorstop <-> (!doorclose && !dooropen) )

[] (dooropen -> !doorclose )

[] (doorclose -> !dooropen )
